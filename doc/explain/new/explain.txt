@router.post(
    "/extract-structure-chunk-and-ingest-v3",
    response_model=IngestResponse,
    summary="Extract → structure → chunk → ingest into Elasticsearch in one call.",
)
async def extract_structure_chunk_and_ingest_v3(
    path: Optional[str] = None,
    out_dir: str = "outputs_tables_only",
    max_tokens: int = 450,
    overlap_tokens: int = 80,
    table_row_chunks: bool = False,
    include_assets: bool = False,
    inline_nodes: bool = True,
    force_on_metadata_change: bool = False,
    es=Depends(get_es_client),
) -> IngestResponse:
    inputs = _resolve_inputs(path)
    resolved_out_dir = _resolve_out_dir(out_dir)

    chunking_output = _run_extract_structure_chunking(
        inputs=inputs,
        out_dir=resolved_out_dir,
        max_tokens=max_tokens,
        overlap_tokens=overlap_tokens,
        table_row_chunks=table_row_chunks,
        include_assets=include_assets,
        inline_nodes=inline_nodes,
    )

    ingest_payload = IngestRequest.model_validate({
        **chunking_output,
        "force_on_metadata_change": force_on_metadata_change,
    })

    return await ingest_records(ingest_payload, es=es)

def _resolve_out_dir(out_dir: str) -> str:
    candidate = Path(out_dir)
    if candidate.is_absolute():
        return str(candidate)
    return str(PROJECT_ROOT / candidate)


def _resolve_inputs(path: Optional[str]) -> list[str]:
    if path and path.strip():
        return [path.strip()]
    return DEFAULT_INPUTS.copy()
