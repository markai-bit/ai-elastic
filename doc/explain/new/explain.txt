# SECTION 1: Embedding Utilities
# ===========================================================================

def get_embedding(text: str, model: str = OPENAI_EMBED_MODEL) -> list:
    """Get embedding for a single text via the AI router service."""
    url = f"{ROUTER_SERVICE_ENDPOINT}/ai-router-service/embedding/serial"
    payload = {"model": model, "data": text}
    response = requests.post(url, json=payload, timeout=HTTP_TIMEOUT_SECONDS)
    if response.status_code in {401, 403}:
        raise RuntimeError(
            "Embedding authentication failed. Check router credentials / environment configuration."
        )
    response.raise_for_status()
    return response.json().get("embedding", [])


def estimate_token_count(text: str) -> int:
    """Approximate token count: ~1 token per 3 characters."""
    return len(text) // 3


def get_embedding_model_info(model: str) -> dict:
    """Fetch model metadata (batch size, token limit) from the router."""
    url = f"{ROUTER_SERVICE_ENDPOINT}/ai-router-service/embedding/model-info"
    params = {"model": model}
    response = requests.get(url, params=params, timeout=HTTP_TIMEOUT_SECONDS)
    if response.status_code in {401, 403}:
        raise RuntimeError(
            "Embedding authentication failed. Check router credentials / environment configuration."
        )
    if response.status_code == 200:
        return response.json()
    return {"max_batch_size": 16, "input_token_limit": 8192}


def get_batch_embeddings(texts: list, model: str = OPENAI_EMBED_MODEL) -> list:
    """
    Generate embeddings for a list of texts in batches.
    Handles token limits, batch sizing, and fallback to serial embedding on errors.
    """
    valid_texts = [t for t in texts if isinstance(t, str) and t.strip()]
    if not valid_texts:
        return []

    model_info = get_embedding_model_info(model)
    max_batch_size = model_info.get("max_batch_size", 16)
    input_token_limit = model_info.get("input_token_limit", 8192)
    url = f"{ROUTER_SERVICE_ENDPOINT}/ai-router-service/embedding/batch"
    headers = {"Content-Type": "application/json", "Accept": "application/json"}
    embeddings = []
    i = 0

    while i < len(valid_texts):
        # Optional progress logging
        if os.getenv("RAG_EMBED_PROGRESS", "false").strip().lower() in {"1", "true", "yes"}:
            logger.info("Embedding progress: %d texts pending...", len(valid_texts) - i)

        # Build a batch that fits within token and size limits
        batch = []
        batch_token_count = 0
        while len(batch) < max_batch_size and i < len(valid_texts):
            text = valid_texts[i]
            tokens = estimate_token_count(text)
            if batch_token_count + tokens > input_token_limit:
                break
            batch.append(text)
            batch_token_count += tokens
            i += 1

        # If a single text exceeds the token limit, truncate it
        if not batch:
            text = valid_texts[i]
            tokens = estimate_token_count(text)
            if tokens > input_token_limit:
                text = text[: input_token_limit * 3]
            batch = [text]
            i += 1

        # Send batch request
        payload = {"model": model, "data_batch": batch}
        response = requests.post(url, json=payload, headers=headers, timeout=HTTP_TIMEOUT_SECONDS)

        if response.status_code in {401, 403}:
            raise RuntimeError(
                "Embedding authentication failed. Check router credentials / environment configuration."
            )

        if response.status_code != 200:
            logger.warning("Batch embedding error: %s %s", response.status_code, response.text)
            # Fallback: serial embedding for each item in the failed batch
            for text in batch:
                try:
                    emb = get_embedding(text, model)
                except Exception as e:
                    logger.warning("Serial embedding error: %s", e)
                    emb = None
                embeddings.append(emb)
        else:
            embeddings.extend(response.json().get("embeddings", [None] * len(batch)))

    return embeddings

def parse_pdf(filepath: str) -> str:
    """Extract raw text from a PDF file using PyPDF2."""
    print(f"\n[1] Parsing PDF: {filepath}")
    text = ""
    with open(filepath, "rb") as file:
        reader = PyPDF2.PdfReader(file)
        for page in reader.pages:
            text += page.extract_text() + "\n"
    return text

# SECTION 3: Text Chunking Strategies
# ===========================================================================

def chunking(text: str, chunk_size: int = 200, overlap: int = 50) -> List[str]:
    """
    Basic fixed-window chunking with word-level overlap.
    Splits text into overlapping windows so context isn't lost.
    """
    print(f"[2] Performing Semantic Chunking (Target Size: {chunk_size} chars)...")
    words = text.replace("\n", " ").split()
    chunks = []
    current_chunk = []
    current_length = 0

    for word in words:
        current_chunk.append(word)
        current_length += len(word) + 1  # +1 for space

        if current_length >= chunk_size:
            chunks.append(" ".join(current_chunk))
            # Keep the last few words for overlap
            overlap_words = current_chunk[-max(1, overlap // 5) :]
            current_chunk = overlap_words
            current_length = sum(len(w) + 1 for w in overlap_words)

    if current_chunk:
        chunks.append(" ".join(current_chunk))

    print(f"    -> Generated {len(chunks)} chunks.")
    return chunks


def _get_nlp():
    """Lazy-load spaCy English model for sentence splitting."""
    global _NLP
    if _NLP is not None:
        return _NLP
    try:
        import en_core_web_sm

        _NLP = en_core_web_sm.load()
    except Exception:
        _NLP = None
    return _NLP


def _split_sentences(text: str) -> list:
    """Split text into sentences using spaCy (preferred) or regex fallback."""
    nlp = _get_nlp()
    if nlp is not None:
        doc = nlp(text)
        return [s for s in (sent.text.strip() for sent in doc.sents) if s]
    # Regex fallback: split on sentence-ending punctuation
    parts = re.split(r"(?<=[.!?])\s+", text or "")
    return [part.strip() for part in parts if part and part.strip()]


def get_openai_embedding(text: str) -> list:
    """Single-text embedding helper (convenience wrapper)."""
    return get_embedding(text, OPENAI_EMBED_MODEL)


def recursive_split(text: str, max_chunk_size: int = 1000) -> List[str]:
    """Fallback chunking: split by paragraphs, then by sentences if too large."""
    paragraphs = text.split("\n\n")
    chunks = []
    for para in paragraphs:
        if len(para) <= max_chunk_size:
            chunks.append(para)
        else:
            chunks.extend(para.split(". "))
    return chunks


def _cos_sim(a: list, b: list) -> float:
    """Compute cosine similarity between two embedding vectors."""
    va = np.asarray(a, dtype=np.float32).reshape(1, -1)
    vb = np.asarray(b, dtype=np.float32).reshape(1, -1)
    return float(cosine_similarity(va, vb)[0, 0])


def _embed_sentences(sentences: list, model: str) -> list:
    """Embed a list of sentences individually."""
    return [get_embedding(s, model) for s in sentences]


def semantic_chunk_text(
    text: str, similarity_threshold: float = 0.75, max_chunk_size: int = 650
) -> List[str]:
    """
    Semantically chunk text by comparing consecutive sentence embeddings.
    Sentences are grouped together as long as:
      - Their cosine similarity exceeds the threshold
      - The chunk size stays under max_chunk_size
    Falls back to recursive_split if embeddings fail.
    """
    sentences = _split_sentences(text)

    if not sentences:
        return []

    embeddings = get_batch_embeddings(sentences, OPENAI_EMBED_MODEL)

    # Defensive: if router returns fewer embeddings (or None entries), fall back
    if (
        not embeddings
        or len(embeddings) != len(sentences)
        or any(e is None for e in embeddings)
    ):
        return recursive_split(text, max_chunk_size=max_chunk_size)

    chunks = []
    current_chunk = []
    current_length = 0

    for i, sentence in enumerate(sentences):
        sentence_len = len(sentence)

        if not current_chunk:
            current_chunk.append(sentence)
            current_length += sentence_len
            continue

        sim = _cos_sim(embeddings[i - 1], embeddings[i])

        if sim < similarity_threshold or current_length + sentence_len > max_chunk_size:
            # Start a new chunk
            chunks.append(" ".join(current_chunk))
            current_chunk = [sentence]
            current_length = sentence_len
        else:
            current_chunk.append(sentence)
            current_length += sentence_len

    if current_chunk:
        chunks.append(" ".join(current_chunk))

    return chunks


def add_overlap(chunks: List[str], overlap_sentences: int = 1) -> List[str]:
    """
    Add overlapping context between consecutive chunks by prepending the
    last N sentences of the previous chunk to the current one.
    """
    new_chunks = []

    for i in range(len(chunks)):
        if i == 0:
            new_chunks.append(chunks[i])
        else:
            prev_chunk = chunks[i - 1]
            prev_sentences = _split_sentences(prev_chunk)
            overlap = " ".join(prev_sentences[-overlap_sentences:])
            new_chunk = overlap + " " + chunks[i]
            new_chunks.append(new_chunk)

    return new_chunks

# SECTION 4: Indexing & Export
# ===========================================================================

def prepare_data_for_indexing(chunks: List[str]) -> List[Dict[str, Any]]:
    """Generate vector embeddings for each chunk and format for Elasticsearch."""
    print("[3] Generating Vector Embeddings using local MiniLM model...")

    embeddings = get_batch_embeddings(chunks, model)

    documents = []
    for i, (chunk, embedding) in enumerate(zip(chunks, embeddings)):
        doc = {
            "chunk_id": f"chunk_{i}",
            "content": chunk,
            "embedding": embedding,
        }
        documents.append(doc)

    return documents


def export_to_json(documents: List[Dict[str, Any]]):
    """Save the indexing payload to a JSON file (embeddings truncated for readability)."""
    print(f"[4] Saving indexing payload format to {EXPORT_JSON_PATH}...")
    with open(EXPORT_JSON_PATH, "w") as f:
        display_docs = []
        for d in documents:
            display_doc = d.copy()
            display_doc["embedding"] = (
                f"[{d['embedding'][0]:.4f}, {d['embedding'][1]:.4f}, ... (384 dimensions)]"
            )
            display_docs.append(display_doc)
        json.dump(display_docs, f, indent=4)


# ===========================================================================
# SECTION 5: Elasticsearch Index Setup & Ingestion
# ===========================================================================

async def setup_elasticsearch_index():
    """Create the Elasticsearch index with dense vector mapping for KNN search."""
    print(f"\n[5] Setting up Elasticsearch Index '{INDEX_NAME}'...")

    # Delete if exists for a clean POC run
    if await es.indices.exists(index=INDEX_NAME):
        await es.indices.delete(index=INDEX_NAME)

    mapping = {
        "mappings": {
            "properties": {
                "chunk_id": {"type": "keyword"},
                "content": {"type": "text"},  # For BM25 Lexical Search
                "embedding": {
                    "type": "dense_vector",   # For Vector KNN Search
                    "dims": 384,              # MiniLM model dimension size
                    "index": True,
                    "similarity": "cosine",   # Distance metric
                },
            }
        }
    }

    await es.indices.create(index=INDEX_NAME, body=mapping)
    print("    -> Index created successfully.")


async def ingest_to_elasticsearch(documents: List[Dict[str, Any]]):
    """Push all document chunks into Elasticsearch in bulk."""
    print(f"[6] Ingesting {len(documents)} documents into Elasticsearch...")
    actions = []
    for doc in documents:
        action = {
            "_op_type": "index",
            "_index": INDEX_NAME,
            "_id": doc["chunk_id"],
            "_source": doc,
        }
        actions.append(action)

    await helpers.async_bulk(es, actions)
    await es.indices.refresh(index=INDEX_NAME)  # Force refresh for immediate searchability
    print("    -> Ingestion complete!")


# ===========================================================================
# SECTION 6: Hybrid Search (BM25 + KNN)
# ===========================================================================

async def hybrid_search(query: str, top_k: int = 10):
    """
    Perform hybrid search combining:
      - BM25 lexical search (exact text matching)
      - KNN vector search (semantic meaning)
    Both are weighted equally (boost=0.5 each).
    """
    print(f"\n[7] Executing Hybrid Search for query: '{query}'")

    # 1. Vectorize the search query
    query_vector = get_embedding(query, model)

    # 2. Build Hybrid Query (Supported in ES 8.x)
    es_query = {
        "size": top_k,
        "query": {
            # LEXICAL (BM25) SEARCH
            "match": {
                "content": {
                    "query": query,
                    "boost": 0.5,  # Weight for text match
                }
            }
        },
        "knn": {
            # VECTOR (KNN) SEARCH
            "field": "embedding",
            "query_vector": query_vector,
            "k": top_k,
            "num_candidates": 50,
            "boost": 0.5,  # Weight for vector meaning
        },
    }

    response = await es.search(index=INDEX_NAME, body=es_query)

    # 3. Display results
    print("\n" + "=" * 40)
    print(" HYBRID SEARCH RESULTS")
    print("=" * 40)
    for i, hit in enumerate(response["hits"]["hits"]):
        score = hit["_score"]
        content = hit["_source"]["content"]
        print(f"Rank {i+1} | Score: {score:.4f}")
        print(f"Content: {content}\n")

# SECTION 7: Main Entry Point

async def main():
    try:
        # Step 0: Ensure we have a PDF to read
        PDF_PATH = "files\\Credit Risk_Exposure Classification.pdf"

        # Step 1 & 2: Parse and Chunk
        raw_text = parse_pdf(PDF_PATH)
        chunks = semantic_chunk_text(raw_text, similarity_threshold=0.75, max_chunk_size=650)
        print(f"\nGenerated {len(chunks)} semantically chunked pieces for indexing.")

        # Step 3 & 4: Embed and Export Format
        documents = prepare_data_for_indexing(chunks)
        export_to_json(documents)
        print(f"\nSample Document for Indexing:\n{json.dumps(documents[0], indent=4)}")

        # Step 5 & 6: Elasticsearch Setup and Ingest
        await setup_elasticsearch_index()
        await ingest_to_elasticsearch(documents)

        # Step 7: Perform Hybrid Search Tests
        await hybrid_search(
            "Article 143(1) of the CRR requires the Prudential Regulation Authority (PRA) "
            "to grant permission to use the Internal Ratings Based (IRB) approach where it "
            "is satisfied that the requirements of Title II Chapter 3 of the CRR are met. "
            "The purpose of this supervisory statement is to provide explanation, where "
            "appropriate, of the PRA's expectations when assessing whether firms meet those "
            "requirements, including in respect of the conservatism applied."
        )

    except Exception as e:
        print(e)
    finally:
        await es.close()
