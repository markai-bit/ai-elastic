failed_chunk_ids: set[str] = set()

if operations:
    try:
        resp = await es.bulk(operations=operations, refresh="wait_for")
    except Exception as exc:  # noqa: BLE001
        log.exception("bulk_ingest_failed", extra={"index": index, "ops": len(operations)})
        raise IndexOperationError(f"Bulk ingest failed: {exc}") from exc

    if resp.get("errors"):
        # Walk every item and collect the ones that actually failed.
        items = resp.get("items", [])
        first_err = None
        for item in items:
            op = item.get("index") or item.get("create") or {}
            if "error" in op:
                failed_chunk_ids.add(op.get("_id"))
                if first_err is None:
                    first_err = op.get("error")

        log.error(
            "bulk_ingest_partial_failure",
            extra={
                "index": index,
                "total_ops": len(operations) // 2,
                "failed_count": len(failed_chunk_ids),
                "first_error": first_err,
            },
        )

        # If literally every write failed, this isn't partial — it's total.
        # Raise so the caller gets a clear 502 instead of a misleading 200.
        if len(failed_chunk_ids) == len(chunks_to_embed):
            raise IndexOperationError(
                f"All {len(failed_chunk_ids)} bulk writes failed. First error: {first_err}"
            )

outcomes: list[RecordOutcome] = []
ingested = skipped = failed = 0
for i, decision in enumerate(decisions):
    if decision["action"] == "write":
        chunk_id = chunk_ids.get(i)
        if chunk_id in failed_chunk_ids:
            outcomes.append(RecordOutcome(
                record_index=i,
                status="failed",
                chunk_id=chunk_id,
                content_hash=decision["hash"],
                reason="Bulk write rejected by Elasticsearch",
            ))
            failed += 1
        else:
            outcomes.append(RecordOutcome(
                record_index=i,
                status=decision["status"],
                chunk_id=chunk_id,
                content_hash=decision["hash"],
            ))
            ingested += 1
    else:
        outcomes.append(RecordOutcome(
            record_index=i,
            status="skipped_duplicate",
            content_hash=decision["hash"],
            reason=decision.get("reason"),
        ))
        skipped += 1

return IngestResponse(
    index=index,
    total=len(request.records),
    ingested=ingested,
    skipped=skipped,
    failed=failed,    # was hardcoded 0 before
    outcomes=outcomes,
)
