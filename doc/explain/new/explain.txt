def _raise_http_from_rag(exc: RagError) -> None:
    """Convert a domain RagError into a FastAPI HTTPException with structured detail."""
    logger.warning(
        "rag_domain_error",
        error_code=exc.error_code,
        error_message=exc.message,
        details=exc.details,
    )
    raise HTTPException(
        status_code=exc.http_status,
        detail={
            "error_code": exc.error_code,
            "message": exc.message,
            "details": exc.details,
        },
    ) from exc

@router.post(
    "/ingestion",
    response_model=IngestResponse,
    summary="Ingest pre-embedded records into Elasticsearch (with hash-based dedup).",
)
async def ingest(
    payload: IngestRequest,
    es=Depends(get_es_client),
) -> IngestResponse:
    logger.info(
        "ingest_request_received",
        records=len(payload.records),
        force_on_metadata_change=payload.force_on_metadata_change,
    )
    try:
        return await ingest_records(payload, es=es)
    except RagError as exc:
        _raise_http_from_rag(exc)

@router.post(
    "/extract-structure-chunk-and-ingest-v3",
    response_model=IngestResponse,
    summary="Extract → structure → chunk → ingest into Elasticsearch in one call.",
)
async def extract_structure_chunk_and_ingest_v3(
    path: Optional[str] = None,
    out_dir: str = "outputs_tables_only",
    max_tokens: int = 450,
    overlap_tokens: int = 80,
    table_row_chunks: bool = False,
    include_assets: bool = False,
    inline_nodes: bool = True,
    force_on_metadata_change: bool = False,
    es=Depends(get_es_client),
) -> IngestResponse:
    inputs = _resolve_inputs(path)
    resolved_out_dir = _resolve_out_dir(out_dir)

    chunking_output = _run_extract_structure_chunking(
        inputs=inputs,
        out_dir=resolved_out_dir,
        max_tokens=max_tokens,
        overlap_tokens=overlap_tokens,
        table_row_chunks=table_row_chunks,
        include_assets=include_assets,
        inline_nodes=inline_nodes,
    )

    ingest_payload = IngestRequest.model_validate({
        **chunking_output,
        "force_on_metadata_change": force_on_metadata_change,
    })

    try:
        return await ingest_records(ingest_payload, es=es)
    except RagError as exc:
        _raise_http_from_rag(exc)

@asynccontextmanager
async def lifespan(_: FastAPI):
    await start_stream_flusher()
    await start_downstream_workers()
    try:
        yield
    finally:
        await stop_stream_flusher()
        await close_es_client()
